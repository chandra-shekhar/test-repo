PHP Sample Token Encrypt Implementation 

function EncryptToken($string) {
        //Key
        $key = "jbngApiPass";

        //Encryption
        $cipher_alg = MCRYPT_TRIPLEDES;

        $iv = mcrypt_create_iv(mcrypt_get_iv_size($cipher_alg,MCRYPT_MODE_ECB), MCRYPT_RAND);

        $encrypted_string = mcrypt_encrypt($cipher_alg, $key, $string, MCRYPT_MODE_ECB, $iv);
        return base64_encode($encrypted_string);
    }

C# Sample Token Encrypt Implementation 


using System;
using System.Security.Cryptography;
using System.Text;

public class Crypto3DES
{
    public Crypto3DES()
    {

    }

    private System.Text.Encoding encoding;

    public string Key
    {
        get
        {
            return "jbngApiPass";
        }
    }

    public System.Text.Encoding Encoding
    {
        get
        {
            if( encoding == null )
            {
                encoding = System.Text.Encoding.UTF8;
            }
            return encoding;
        }

        set
        {
            encoding = value;
        }
    }

    public string EncryptToken( string strString )
    {
        DESCryptoServiceProvider DES = new DESCryptoServiceProvider();

        DES.Key = Encoding.GetBytes( this.Key );
        DES.Mode = CipherMode.ECB;
        DES.Padding = PaddingMode.Zeros;

        ICryptoTransform DESEncrypt = DES.CreateEncryptor();

        byte[] Buffer = encoding.GetBytes(strString);

        return Convert.ToBase64String(DESEncrypt.TransformFinalBlock(Buffer, 0, Buffer.Length));
    }

    5. Get Order items by Status

    Description

    Get List of order items since specified date-time & Status. 

    http://api.jabong.com/ccapi/order/orderitemsByStatus/dt/{datetime}/?status={export}&authToken={encryptedToken}
    datetime
    Date & Time for Order Status Sprint by Updated Date
    Example: 2012-03-20 16:23:45
    {export}
    Export Status which is 3 in current system
    Example: 3 (mandatory)
    ID name description
    1 new new order
    2 payment_pending we wait for money after authorization
    3 exportable will be exported in the next run of the export cron job
    4 exported the item was sent to the fulfillment provider
    5 shipped the item was sent to the customer
    6 closed the item was successfully processed
    7 clarify_not_shipped the fulfillment provider did not ship the order
    8 returned the item was returned by the customer
    9 canceled the item was canceled by the costumer support or cronjob (before we received money)
    10 invalid order is not valid (can be removed from database)
    11 return_denied the return of an item was denied
    12 refunded the returned item was refunded
    13 retransfer the item was canceled and the money needs to be retransfered
    14 clarify_refund_failed the refunding failed
    15 clarify_payment_error The customer paid not enough or too much
    16 clarify_not_paid the item was shipped, but no payment was received
    17 money_received the payment was received from the provider
    18 returned_payment_pending the item was returned, but payment is still pending
    19 rejected the customer did not accept the delivery
    20 return_denied_payment_pending the item was returned, but payment is still pending but return is denied
    21 export_in_progress order export failed and has to be checked
    22 clarify_failed_export NULL
    23 declined order is declined by the customer care
    24 pending order needs interaction by customer care
    25 cancelled_cc Cancelled by Customer Care Team
    26 exportable_cancelled_cust Exportable cancelled by Customer
    27 exported_cancelled_cust Exported cancelled by Customer
    28 cancelled_wh Cancelled by WareHouse Team
    29 cancelled_payment_error Cancelled Payment Error
    30 refunded_after_cancellation Refunded after cancellation
    31 refunded_after_shipped Refunded after shipped
    32 refunded_after_returned Refunded after returned

    15. Get Order ID by item status, date-time and shipment type

    Description

    datetime
    Updated Date & Time for order status.
    Example
    2014-03-04 12:40:09
    status
    status code.
    Example
    3
    shipmentType
    Shipment Type.
    Example
    1 in case of own warehouse and 4 in case of marketplace.
    If no shipment type is passed, then system will consider all shipment types. More than one shipment type can be passed with hyphen like “1-2”.
